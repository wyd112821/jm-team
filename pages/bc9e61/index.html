# 表情组件

表情 

<template>
    <div>
        <a-button type="link" @click="visible=!visible">表情</a-button>
        <face :visible="visible" @closeEvent="closeModal" @addEvent="addFace"></face>
    </div>
</template>
<script>
   export default {
       data(){
           return{
               visible: false,
           }
       },
       methods:{
           closeModal () {
               this.visible = false;
           },
           // 添加表情
           addFace (item) {
               console.log(item)
               this.visible = false;
           },
       }
   }
</script>
