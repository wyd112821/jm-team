# 评分组件

<template>
    <div>
        <a-button type="link" @click="visible=true">评分</a-button>
        <default-modal title="标题"
               :visible="visible"
               :width="480"
               :maskClosable="false"
               centered
               okText="确 定"
               cancelText="取 消"
               @ok="onSumit"
               @cancel="onCancel"
        >
            <!--难度分-->
            <template>
                <div class="score-item">
                    <div class="score-item-left">难度分</div>
                    <a-rate v-model="curNd" :tooltips="numList" :count="10"/>
                    <div class="score-item-text">{{ curNd }}分</div>
                </div>
            </template>
            <!--质量分-->
            <template>
                <div class="score-item">
                    <div class="score-item-left">质量分</div>
                    <a-rate v-model="curZl" :tooltips="numList" :count="10"/>
                    <div class="score-item-text">{{ curZl }}分</div>
                </div>
            </template>
            <!--满意度-->
            <template>
                <div class="score-item">
                    <div class="score-item-left">满意度</div>
                    <a-rate v-model="curMy" :tooltips="numList" :count="10"/>
                    <div class="score-item-text">{{ curMy }}分</div>
                </div>
            </template>
            <!--评论清单-->
            <template>
                <div class="comment-list">
                <span class="comment-item"
                      :class="curTags.includes(item) ? 'on' : ''"
                      @click="selectName(item)"
                      v-for="item in commentList"
                      :key="item"
                >{{ item }}</span>
                </div>
            </template>
        </default-modal>
    </div>
    
</template>
<script>
    const numList=[
        '1分',
        '2分',
        '3分',
        '4分',
        '5分',
        '6分',
        '7分',
        '8分',
        '9分',
        '10分',
    ];
    const commentList=[ '效率超高效',
        '沟通协作顺畅',
        '响应速度快',
        '效果体验超预期',
        '需求定位准确',
        '进度神准时',
        '创新度、效果极佳',
        '工作认真负责',]
    export default {
        data() {
            return {
                visible: false,
                curNd: 0,
                curZl:0,
                curMy:0,
                numList,
                commentList,
                curTags:[]
            }
        },
        methods: {
            // 关闭弹窗
            onCancel() {
                this.visible = false
            },
            // 提交评分
            onSumit() {
                this.visible = false
            },
            selectName(item){
                let {curTags}=this
                if(curTags.includes(item)){
                    curTags.splice(curTags.indexOf(item),1)
                }else {
                    curTags.push(item)
                }
            }
        }
    }
</script>
